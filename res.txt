def reverse_string(s):
    result = ""
    for i in range(len(s) - 1, -1, -1):
        result += s[i]
    return result

def add(a, b):
    carry = 0
    result = []
    while a or b or carry:
        digit_sum = (a % 10 + b % 10 + carry) % 10
        carry = (a % 10 + b % 10 + carry) // 10
        result.append(digit_sum)
        a //= 10
        b //= 10
    return ''.join(map(str, reversed(result)))
